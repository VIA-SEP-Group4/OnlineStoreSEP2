
create table eshop.users(
	user_id serial,
	username varchar(20),
	pass varchar(20),
	email varchar(50),
	firstname varchar(20),
	lastname varchar(20)
);

alter table eshop.users add constraint pk_users primary key (user_id);
alter table eshop.users add constraint unique_username UNIQUE(username);


select * from eshop.users;
